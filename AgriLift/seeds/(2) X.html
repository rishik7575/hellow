    // Cart Functionality
    let cart = JSON.parse(localStorage.getItem('cart')) || [];

    function updateCart() {
        const cartCount = cart.reduce((sum, item) => sum + item.quantity, 0);
        document.getElementById('cartCount').textContent = cartCount;
        localStorage.setItem('cart', JSON.stringify(cart));
        updateCartDisplay();
    }

    function addItemToCart(productName, price) {
        const product = cart.find(item => item.name === productName);

        if (product) {
            product.quantity += 1;
        } else {
            cart.push({
                name: productName,
                price: price,
                quantity: 1
            });
        }
        updateCart();
        showToast();
    }

    // Add to Cart
    document.querySelectorAll('.add-to-cart').forEach(button => {
        button.addEventListener('click', () => {
            const productCard = button.closest('.product-card');
            const productName = productCard.querySelector('h3').textContent;
            const price = parseFloat(productCard.querySelector('span').textContent.slice(1));
            addItemToCart(productName, price);
        });
    });

    // Cart Display
    function updateCartDisplay() {
        const cartItems = document.getElementById('cartItems');
        cartItems.innerHTML = '';

        let total = 0;
        cart.forEach((item, index) => {
            total += item.price * item.quantity;
            cartItems.innerHTML += `
                <div class="cart-item flex justify-between items-center mb-2">
                    <div>
                        <h4 class="font-semibold">${item.name}</h4>
                        <p class="text-sm">$${item.price.toFixed(2)} x ${item.quantity}</p>
                    </div>
                    <div class="flex items-center space-x-2">
                        <button class="change-quantity px-2 rounded bg-gray-100" data-index="${index}" data-change="-1">-</button>
                        <span>${item.quantity}</span>
                        <button class="change-quantity px-2 rounded bg-gray-100" data-index="${index}" data-change="1">+</button>
                        <button class="remove-item px-2 rounded bg-red-100 text-red-600" data-index="${index}">Remove</button>
                    </div>
                </div>
            `;
        });

        document.getElementById('cartTotal').textContent = `$${total.toFixed(2)}`;
    }

    // Change Quantity
    document.addEventListener('click', function (event) {
        if (event.target.classList.contains('change-quantity')) {
            const index = parseInt(event.target.getAttribute('data-index'));
            const change = parseInt(event.target.getAttribute('data-change'));

            if (cart[index]) {
                if (change === -1 && cart[index].quantity > 1) {
                    cart[index].quantity -= 1;
                } else if (change === 1) {
                    cart[index].quantity += 1;
                }
                updateCart();
            }
        }
    });

    // Remove Item from Cart
    document.addEventListener('click', function (event) {
        if (event.target.classList.contains('remove-item')) {
            const index = parseInt(event.target.getAttribute('data-index'));
            if (cart[index]) {
                cart.splice(index, 1); // Remove the item from the array
                updateCart();
            }
        }
    });
